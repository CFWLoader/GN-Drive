\documentclass{article}

\usepackage{amsmath}

\begin{document}

% Matrix without brackets.
$\begin{matrix}
1 & 0\\
0 & 1
\end{matrix}$

 % Matrix with plain [] symbols.
[
$\begin{matrix}						% Matrix should begin with a dollar symbol and end with a dollar symbol.
1 & 0\\
0 & 1
\end{matrix}$
]

% Matrix with scaling.
$\left[
\begin{matrix}
1 & 0\\
0 & 1
\end{matrix}
\right]$

% Scaling for fractions and other expressions.
\begin{equation}
\left(\frac{1}{\sqrt{x}}\right)
\end{equation}

\end{document}
